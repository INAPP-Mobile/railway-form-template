Embed Code

<form action="{{ request.base_url }}form/{{ forms[0]['slug'] }}" method="post">
{% for field in (forms[0].fields or []) %}
  {% if field.type == 'textarea' %}
  <textarea name="{{ field.name }}" placeholder="{{ field.label }}"{% if field.required %} required{% endif %}></textarea>
  {% elif field.type == 'select' %}
  <select name="{{ field.name }}"{% if field.required %} required{% endif %}>
    <option value="">Select {{ field.label }}</option>
    {% for opt in field.options %}
    <option value="{{ opt }}">{{ opt }}</option>
    {% endfor %}
  </select>
  {% elif field.type == 'checkbox' %}
  <label><input type="checkbox" name="{{ field.name }}" value="yes"> {{ field.label }}</label>
  {% else %}
  <input type="{{ field.type }}" name="{{ field.name }}" placeholder="{{ field.label }}"{% if field.required %} required{% endif %}>
  {% endif %}
{% endfor %}
  <script src="https://cdn.jsdelivr.net/npm/cap-widget"></script>
  <cap-widget required
    data-cap-api-endpoint="{{ settings.cap_endpoint or 'https://your-cap.railway.app/YOUR_SITE_KEY/' }}">
  </cap-widget>
  <button type="submit">Send</button>
</form>
